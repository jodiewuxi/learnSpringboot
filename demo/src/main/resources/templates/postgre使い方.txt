CREATE文
CREATE TABLE readbook(
   book_id serial PRIMARY KEY,
   bookname VARCHAR (50) UNIQUE NOT NULL,
   bookwriter VARCHAR (50) NOT NULL,
   comment VARCHAR (355) UNIQUE NOT NULL,
   created_on TIMESTAMP NOT NULL,
   last_login TIMESTAMP
);
